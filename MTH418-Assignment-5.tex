\begin{enumerate}
    \item \textbf{(Review Problem)} Let $X_{1}, \dots, X_{n}$ be a random sample from p.m.f./p.d.f. $f_{\theta}, \theta \in \Theta$.
    In each of the following cases, find MME and MLE of $\theta$. Also, determine if they are functions of a minimal sufficient statistic:
    \begin{enumerate}[(i)]
        \item $X_{1} \sim N(\theta, \sigma_{0}^{2})$, $\Theta = \mathbb{R}$, $\sigma_{0}$ is a known positive constant;
        \item $X_{1} \sim N(\mu_{0}, \theta^{2})$, $\Theta = (0, \infty)$, $\mu_{0}$ is a known real constant;
        \item $X_{1} \sim Exp(\theta, \sigma_{0})$, $\Theta = \mathbb{R}$, $\sigma_{0}$ is a known positive constant;
        \item $X_{1} \sim Exp(\mu_{0}, \theta)$, $\Theta = (0, \infty)$, $\mu_{0}$ is a known real constant;
        \item $X_{1} \sim U(\theta - \sigma_{0}, \theta + \sigma_{0})$, $\Theta = \mathbb{R}$, $\sigma_{0}$ is a known positive constant;
        \item $X_{1} \sim U(\mu_{0} - \theta, \mu_{0} + \theta)$, $\Theta = (0, \infty)$, $\mu_{0}$ is a known real constant;
        \item $f(x) = \frac{\alpha_{0} x^{\alpha_{0}-1}}{\theta^{\alpha_{0}}} (0 < x < \theta)$, $\Theta = (0, \infty)$, $\alpha_{0}$ is a known positive constant;
        \item $f_{\theta}(x) = \alpha_{0} \frac{\theta^{\alpha_{0}}}{x^{\alpha_{0}+1}} I(x > \theta)$, $\Theta = (0, \infty)$, $\alpha_{0}$ is a known positive constant;
        \item $X_{1} \sim Bin(m_{0}, \theta)$, $\Theta = (0, 1)$, $m_{0}$ is a known positive integer;
        \item $X_{1} \sim Bin(\theta, p_{0})$, $\Theta = \{1, 2, \dots\}$, $p_{0} \in (0, 1)$ is a known constant;
        \item $X_{1} \sim Poisson(\theta)$, $\Theta = (0, \infty)$;
        \item $X_{1} \sim Gamma(\theta, \alpha_{0})$, $\Theta = (0, \infty)$, $\alpha_{0}$ is a known positive constant;
        \item $X_{1} \sim Gamma(\sigma_{0}, \theta)$, $\Theta = (0, \infty)$, $\sigma_{0}$ is a known positive constant;
        \item $X_{1} \sim DEXP(\theta, \sigma_{0})$, $\Theta = \mathbb{R}$, $\sigma_{0}$ is a known positive constant;
        \item $X_{1} \sim DEXP(\mu_{0}, \theta)$, $\Theta = (0, \infty)$, $\mu_{0}$ is a known real constant;
    \end{enumerate}

    \item Find two examples where the MLE is not unique.

    \item Let $X \sim Gamma(1, \alpha)$, $\alpha > 0$. Show that $E(\ln X) = \psi(\alpha)$, where $\psi(\alpha) = \frac{d}{d\alpha} \ln \Gamma(\alpha) = \frac{\Gamma'(\alpha)}{\Gamma(\alpha)}$, $\alpha > 0$ is called the digamma function. Show that:
    \begin{enumerate}[(i)]
        \item $\psi(\alpha) < \ln \alpha$, $\alpha > 0$;
        \item $\psi(\alpha)$ is an increasing function on $(0, \infty)$.
    \end{enumerate}

    \item 
    \begin{enumerate}[(i)]
        \item Let $X_{1}, \dots, X_{n}$ be a random sample from $N(\mu, \sigma^{2})$ where $\underline{\theta} = (\mu, \sigma) \in \mathbb{R} \times (0, \infty) = \Theta$. Find the MLE and the MME of $\tau(\underline{\theta}) = P_{\underline{\theta}}(X_{1} > 1)$;
        \item Let $X_{1}, \dots, X_{n}$ be a random sample from $Gamma(\sigma, \alpha)$, where $\underline{\theta} = (\sigma, \alpha) \in (0, \infty) \times (0, \infty) = \Theta$. Find the MLEs and MMEs of $\underline{\theta}$ and $\tau(\underline{\theta}) = Var_{\underline{\theta}}(X_{1})$;
        \item Let $X_{1}, \dots, X_{n}$ be a random sample from $DEXP(\mu, \sigma)$, $\underline{\theta} = (\mu, \sigma) \in \mathbb{R} \times (0, \infty)$. Find the MLE and the MME of $\underline{\theta}$.
    \end{enumerate}

    \item Let $X_{1}, X_{2}$ be a random sample from a p.d.f. $f_{\theta}(x) = \frac{2}{\theta^{2}}(\theta - x) I(0 < x < \theta)$, $\theta \in (0, \infty) = \Theta$. Find the MME and the MLE of $\theta$. Are they functions of a minimal sufficient statistic?

    \item Let $X_{1}, \dots, X_{n}$ be a random sample from p.m.f./p.d.f. $f_{\theta}, \theta \in \Theta$. In each of the following cases, find MME and MLE of $\theta$. Also determine if they are functions of a minimal sufficient statistic:
    \begin{enumerate}[(i)]
        \item $X_{1} \sim N(\theta, \sigma_{0}^{2})$, $\Theta = (a, b)$, $a, b$ ($a < b$) and $\sigma_{0}$ are known positive constants;
        \item $X_{1} \sim Exp(\theta, \sigma_{0})$, $\Theta = (a, b)$, $a, b$ ($a < b$) and $\sigma_{0}$ are known positive constants;
        \item $X_{1} \sim Bin(m_{0}, \theta)$, $\Theta = [\frac{1}{2}, 1)$, $m_{0}$ is a known positive integer;
        \item $X_{1} \sim U(\theta_{1}, \theta_{2})$, $\underline{\theta} = (\theta_{1}, \theta_{2})$, $\Theta = \{(x, y) \in \mathbb{R}^{2} : x < y\}$;
        \item $X_{1} \sim U(\theta_{1} - \theta_{2}, \theta_{1} + \theta_{2})$, $\underline{\theta} = (\theta_{1}, \theta_{2})$, $\Theta = \{(x, y) \in \mathbb{R}^{2} : -\infty < x < \infty, y > 0\}$.
    \end{enumerate}

    \item Let $X_{1}, \dots, X_{n}$ be a random sample from $Exp(0, \theta)$, $\theta \in \Theta = (0, \infty)$. Let $c$ be a positive constant. What is being observed is only those $X_{i}$s whose values are less than $c$. Suppose that the observed values are $Y_{1}, \dots, Y_{m}$ and remaining $(n - m)$ observations that exceed $c$ are not observed. Find the MLE of $\theta$.

    \item Prove the following inequalities for any non-degenerate random variable $X$: 
    \begin{enumerate}[(i)]
        \item $E(e^{X}) > e^{E(X)}$; 
        \item $E(X) E(\frac{1}{X}) > 1$, provided $P(X > 0) = 1$;
        \item $E(\ln X) < \ln E(X)$, provided $P(X > 0) = 1$. 
    \end{enumerate}
    Hence prove the AM-GM-HM inequality:
    \[ \sum_{i=1}^{n} a_{i} w_{i} > \prod_{i=1}^{n} a_{i}^{w_{i}} > \frac{1}{\sum_{i=1}^{n} \frac{w_{i}}{a_{i}}} \]
    for any positive constants $w_{1}, \dots, w_{n}, a_{1}, \dots, a_{n}$, such that $\sum_{i=1}^{n} w_{i} = 1$ and not all $a_{i}$s are the same.

    \item Let $X_{1}, \dots, X_{n}$ be a random sample from $N(\theta, 1)$, $\theta \in \mathbb{R} = \Theta$. For estimating $\theta$ under a loss function, consider the randomized estimator $\delta(\underline{x}) \sim N(\frac{x_{1} + x_{2}}{2}, 1)$.
    \begin{enumerate}[(i)]
        \item Find a randomized estimator that is based on a minimal sufficient statistic and has the same risk function as $\delta(\underline{x})$;
        \item Under the squared error loss function find a non-randomized estimator better than $\delta$.
    \end{enumerate}

    \item Let $X_{1}, \dots, X_{n}$ ($n \ge 2$) be a random sample from $Exp(0, \theta)$, $\theta \in (0, \infty) = \Theta$. For estimating $\theta$ under a loss function, consider the randomized estimator $\delta(\underline{x}) \sim U(\frac{n-1}{n} \bar{x}_{n-1}, \bar{x}_{n-1})$, where $\bar{x}_{n-1} = \frac{1}{n-1} \sum_{i=1}^{n-1} x_{i}$.
    \begin{enumerate}[(i)]
        \item Find a randomized estimator based on a minimal sufficient statistic having the same risk function as $\delta(\underline{X})$;
        \item Under the squared error loss function find a non-randomized estimator better than $\delta$.
    \end{enumerate}

    \item Let $X_{1}, \dots, X_{n}$ ($n \ge 2$) be a random sample from $N(\theta, 1)$ where $\theta \in \mathbb{R} = \Theta$ is the unknown parameter. For estimating $\theta$, suppose that the loss function is either the absolute error loss function or the squared error loss function.
    \begin{enumerate}[(i)]
        \item Find an estimator dominating the estimator $\delta_{1}(\underline{X}) = \sum_{i=1}^{n} \alpha_{i} X_{i}$, where $\alpha_{1}, \dots, \alpha_{n}$ are positive constants such that $\sum_{i=1}^{n} \alpha_{i} = 1$ and not all $\alpha_{i}$s are the same;
        \item Let $n = 2m + 1$. Find a randomized estimator, based on $\bar{X}$, which has the same risk function as the sample median $\delta_{2}(\underline{X}) = X_{(m+1)}$;
        \item Find an estimator better than $\delta_{2}(\underline{X}) = X_{(m+1)}$.
    \end{enumerate}

    \item Let $X \sim Bin(n, \theta)$, $\theta \in (0, 1) = \Theta$. For estimating $\theta$ under the squared error loss function, consider the randomized estimator $\delta$, such that $P(\delta(x) = \frac{x}{n}) = P(\delta(x) = \frac{1}{2}) = \frac{1}{2}$. Find a nonrandomized estimator better than $\delta$. Also calculate the risk functions of the two estimators.

    \item Let $X_{1}, \dots, X_{n}$ be a random sample from $U(0, \theta)$, where $\theta \in (0, \infty) = \Theta$ is the unknown parameter. Consider estimation of $\theta$ under the absolute error loss function.
    \begin{enumerate}[(i)]
        \item Find an estimator, based on a minimal sufficient statistic, that is better than $\delta_{0}(\underline{X}) = \bar{X}$;
        \item Find an estimator, based on a minimal sufficient statistic, that is better than $\delta_{1}(\underline{X}) = X_{(n-1)}$.
    \end{enumerate}
\end{enumerate}
